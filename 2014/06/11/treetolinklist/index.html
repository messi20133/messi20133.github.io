将二叉树变成链表


将二叉树变成链表
二叉树如图所示
1
/ \
2 5
/ \ \
3 4 6
要改成1-2-3-4-5-6（也就是前序遍历）
思路：
从根节点一直走左子树，如果右子树不为空，则将右子树放入栈中。
代码如下：

package Tree;
import java.util.LinkedList;
import java.util.Stack;
public class FlattenTree {
public static void flatten(Node root){
    Stack<Node> stack=new Stack<Node>();
    Node p=root;
    while(p!=null||!stack.empty()){
        if(p.right!=null)
            stack.push(p.right);
        if(p.left!=null){
            p.right=p.left;
            p.left=null;
        }else if(!stack.empty()){
            Node temp=stack.pop();
            p.right=temp;
        }
        p=p.right;
    }
}
public static void showList(Node root){
    if(root==null){
        return;
    }
    Node p=root;
    while(p!=null){
        System.out.print(p.value+"->");
        p=p.right;
    }
    System.out.println();        
}
public static void main(String[] args) {
    Node root=new Node(1);
    Node n2=new Node(2);
    Node n3=new Node(3);
    Node n4=new Node(4);
    Node n5=new Node(5);
    root.left=n2;
    root.right=n5;
    n2.left=n3;
    n2.right=n4;
    flatten(root);
    showList(root);

}
}
